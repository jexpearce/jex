import pandas as pd
import numpy as np

fruitvegseafood = pd.read_csv('FruitVeg&Seafood.csv')
import matplotlib.pyplot as plt

seafood = fruitvegseafood[fruitvegseafood["Category"] == "Seafood"]

Calorie Distribution in Seafood:

All of these are exactly 85 grams 
calories = seafood[["Food and Serving", "Calories"]]

sorted_calories = calories.sort_values(by='Calories', ascending=True)

ax = sorted_calories.plot.barh(0,
                               1,

%RDI of Iron in Seafood: 

iron = seafood[["Food and Serving","Iron %"]]
filterediron = iron[iron["Iron %"] != 0] #Filters out fish with 0 iron
sorted_iron = filterediron.sort_values(by='Iron %', ascending = True)

ax = sorted_iron.plot.barh(0,1, color = "slategrey", title = "Recommended Daily Intake of Iron in Seafood (%)")

ax.set_xlabel("Iron %")
ax.set_ylabel("Seafood")
plt.text(21, 0, '(Does not include Flounder or Catfish as they have 0 Iron)', fontsize = 10)
