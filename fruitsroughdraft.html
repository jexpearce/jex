Before You Read:

  • %RDI by definition is the average daily dietary intake level that is sufficient to meet the nutrient requirements of nearly all (97â€“98 percent) healthy individuals. (A 2000 calorie diet is recommended).

  • The weight of all the fruits in grams is not constant, as it the data in the database was sorted by how much the average person would eat as a sufficient snack. Ex. Two kiwis, one avocado, 3/4 cup of grapes, etc. 

#Importation
import pandas as pd
import numpy as np
fruitvegseafood = pd.read_csv('FruitVeg&Seafood.csv')
import matplotlib.pyplot as plt

#Creating a subset of the database called fruits
fruits = fruitvegseafood[fruitvegseafood["Category"] == "Fruit"] 

calories = fruits[["Food and Serving", "Calories"]]

sorted_calories = calories.sort_values(by='Calories', ascending=True)

ax = sorted_calories.plot.barh(0,
                               1,
                               color="mediumseagreen",
                               title="Calories in Fruits")

ax.set_xlabel("Calories")
ax.set_ylabel("Fruit")

vitaminc = fruits[["Food and Serving", "Vitamin C %"]]

sorted_vitaminc = vitaminc.sort_values(by='Vitamin C %', ascending=True)

ax = sorted_vitaminc.plot.barh(
    0,
    1,
    color="salmon",
    title="Recommended Daily Intake of Vitamin C in Fruits (%)")

ax.set_xlabel("Vitamin C %")
ax.set_ylabel("Fruit")

potassium = fruits[["Food and Serving", "Potassium %"]]

sorted_potassium = potassium.sort_values(by='Potassium %', ascending=True)

ax = sorted_potassium.plot.barh(
    0,
    1,
    color="teal",
    title="Recommended Daily Intake of Potassium in Fruit (%)")

ax.set_xlabel("Potassium %")
ax.set_ylabel("Fruit")

vitamina = fruits[["Food and Serving", "Vitamin A %"]]
filteredvitamina = vitamina[vitamina["Vitamin A %"] != 0]
sorted_vitamina = filteredvitamina.sort_values(by='Vitamin A %',
                                               ascending=True)

ax = sorted_vitamina.plot.barh(
    0,
    1,
    color="goldenrod",
    title="Recommended Daily Intake of Vitamin A in Fruit (%)")

ax.set_xlabel("Vitamin A %")
ax.set_ylabel("Fruit")
plt.text(65, 0, '(Does not include fruits with 0 Vitamin A)', fontsize=10)
